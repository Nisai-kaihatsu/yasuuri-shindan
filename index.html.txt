    function diagnose() {
        const income = parseFloat(document.getElementById('income').value);
        const ot = parseFloat(document.getElementById('overtime').value) || 0;
        const sleep = parseFloat(document.getElementById('sleep').value) || 7;
        const myTime = parseFloat(document.getElementById('mytime').value) || 0;
        const boss = parseInt(document.getElementById('boss').value);
        const depress = parseInt(document.getElementById('depress').value);
        const age = parseInt(document.getElementById('age').value);

        // バリデーション
        if (isNaN(income) || income <= 0) {
            alert("年収をちゃんと入力するお！0円はさすがに嘘だろｗ");
            return;
        }

        // 1. 搾取スコア (時給ベース)
        let baseWage = age < 25 ? 1200 : (age < 35 ? 1600 : 2000);
        const totalWorkHours = 2000 + (ot * 12);
        const hourly = (income * 10000) / totalWorkHours;
        const wageScore = Math.min(40, Math.max(0, (1 - hourly / baseWage) * 40));

        // 2. 幸福欠如スコア
        const happinessScore = Math.min(30, Math.max(0, 30 - (myTime * 0.75) + (Math.max(0, 6 - sleep) * 3)));

        // 3. 精神摩耗スコア
        const mentalScore = Math.min(30, boss + depress);

        // 合計とソフト補正 (極端な値を出にくくする)
        let totalScore = wageScore + happinessScore + mentalScore;
        totalScore = Math.round(totalScore * 0.85 + 7); // 中央(50点付近)に寄せる
        totalScore = Math.max(0, Math.min(100, totalScore));

        // 表示処理
        document.getElementById('input-view').style.display = 'none';
        document.getElementById('result-view').style.display = 'block';
        document.getElementById('score-num').innerText = totalScore;

        const rankEl = document.getElementById('rank-display');
        if(totalScore >= 85) {
            rankEl.innerText = "【逝ってよし認定】"; rankEl.className = "rank-label rank-high";
        } else if(totalScore >= 60) {
            rankEl.innerText = "【社畜中級】"; rankEl.className = "rank-label rank-mid";
        } else {
            rankEl.innerText = "【まだマターリ可能】"; rankEl.className = "rank-label rank-low";
        }

        // 煽りコメント表示
        const cIdx = Math.floor(Math.random() * jokeComments.length);
        document.getElementById('comment-box').innerText = jokeComments[cIdx];

        // AA表示
        const aaBox = document.getElementById('aa-box');
        aaBox.innerText = totalScore >= 85 ? aaList[Math.floor(Math.random() * aaList.length)] : " ∧＿∧\n（　´∀｀）＜ ぬるぽ";

        // BBS蓄積 (最大20件程度)
        postCount++;
        const list = document.getElementById('bbs-list');
        const shortComment = jokeComments[cIdx].substring(0, 15) + "...";
        const newPost = `<div class="bbs-post">${postCount}: <b>名無しさん＠診断中</b>：スコア${totalScore}点<br>${shortComment}</div>`;
        list.innerHTML = newPost + list.innerHTML;

        renderChart(wageScore, happinessScore, mentalScore);
    }

    function renderChart(w, h, m) {
        const ctx = document.getElementById('myChart').getContext('2d');
        if(myChart) myChart.destroy();
        myChart = new Chart(ctx, {
            type: 'radar',
            data: {
                labels: ['搾取度', '幸福欠如', '精神摩耗'],
                datasets: [{ 
                    data: [w*2.5, h*3.3, m*3.3], 
                    backgroundColor: 'rgba(255,0,255,0.4)', 
                    borderColor: '#f0f',
                    borderWidth: 2
                }]
            },
            options: {
                scales: { 
                    r: { 
                        grid: { color: '#0f0' }, 
                        angleLines: { color: '#0f0' }, 
                        ticks: { display: false }, 
                        min: 0, max: 100 
                    }
                },
                plugins: { legend: { display: false } },
                maintainAspectRatio: false
            }
        });
    }
